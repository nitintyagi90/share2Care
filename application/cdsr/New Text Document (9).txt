function status(id){
var urls = $("#url").val();
var status = $("#aprove_"+id).val();
var ename = $("#ename").val();
var email = $("#email").val();
var mobile = $("#mobile").val();
$.ajax({
type: "POST",
url: urls+"Admin/status",
data: {status:status,id:id,ename:ename,email:email,mobile:mobile},
cache: false,
success: function(result){
location.reload();
}
});
}
